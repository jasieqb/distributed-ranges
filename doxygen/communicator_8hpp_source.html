127 template <rng::contiguous_range R>
128 void alltoallv(const R &sendbuf, const std::vector<std::size_t> &sendcnt,
129 const std::vector<std::size_t> &senddsp, R &recvbuf,
130 const std::vector<std::size_t> &recvcnt,
131 const std::vector<std::size_t> &recvdsp) {
132 using valT = typename R::value_type;
133
134 assert(rng::size(sendcnt) == size_);
135 assert(rng::size(senddsp) == size_);
136 assert(rng::size(recvcnt) == size_);
137 assert(rng::size(recvdsp) == size_);
138
139 std::vector<int> _sendcnt(size_);
140 std::vector<int> _senddsp(size_);
141 std::vector<int> _recvcnt(size_);
142 std::vector<int> _recvdsp(size_);
143
144 rng::transform(sendcnt, _sendcnt.begin(),
145 [](auto e) { return e * sizeof(valT); });
146 rng::transform(senddsp, _senddsp.begin(),
147 [](auto e) { return e * sizeof(valT); });
148 rng::transform(recvcnt, _recvcnt.begin(),
149 [](auto e) { return e * sizeof(valT); });
150 rng::transform(recvdsp, _recvdsp.begin(),
151 [](auto e) { return e * sizeof(valT); });
152
153 MPI_Alltoallv(rng::data(sendbuf), rng::data(_sendcnt), rng::data(_senddsp),
154 MPI_BYTE, rng::data(recvbuf), rng::data(_recvcnt),
155 rng::data(_recvdsp), MPI_BYTE, MPI_COMM_WORLD);
156 }
